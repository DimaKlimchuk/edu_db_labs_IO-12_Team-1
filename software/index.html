# SQL-скрипт для створення на початкового наповнення бази даних

-- MySQL Workbench Forward Engineering

SET @OLD_UNIQUE_CHECKS=@@UNIQUE_CHECKS, UNIQUE_CHECKS=0;
SET @OLD_FOREIGN_KEY_CHECKS=@@FOREIGN_KEY_CHECKS, FOREIGN_KEY_CHECKS=0;
SET @OLD_SQL_MODE=@@SQL_MODE, SQL_MODE='ONLY_FULL_GROUP_BY,STRICT_TRANS_TABLES,NO_ZERO_IN_DATE,NO_ZERO_DATE,ERROR_FOR_DIVISION_BY_ZERO,NO_ENGINE_SUBSTITUTION';

DROP SCHEMA IF EXISTS `survey_of_experts` ;

-- ---------------Створення бази даних---------------------
CREATE SCHEMA IF NOT EXISTS `survey_of_experts` DEFAULT CHARACTER SET utf8 ;
USE `survey_of_experts` ;
-- --------------------------------------------------------

-- ---------------Структура таблиці permission-------------------
DROP TABLE IF EXISTS `survey_of_experts`.`permission` ;

CREATE TABLE IF NOT EXISTS `survey_of_experts`.`permission`(
    `id` INT NOT NULL AUTO_INCREMENT,
    `name` VARCHAR(45) NOT NULL,
    `text` VARCHAR(45) NOT NULL,
    PRIMARY KEY (`id`)
)ENGINE = InnoDB;

-- ------------------------------------------------------------


-- ---------------Структура таблиці user-------------------
DROP TABLE IF EXISTS `survey_of_experts`.`user` ;

CREATE TABLE IF NOT EXISTS `survey_of_experts`.`user`(
    `id` INT NOT NULL AUTO_INCREMENT,
    `username` VARCHAR(20) NOT NULL,
    `email` VARCHAR(45) NOT NULL,
    `password` VARCHAR(20) NOT NULL,
    `permission_id` INT NOT NULL,
    PRIMARY KEY (`id`),
    CONSTRAINT `fk_user_permission`
		FOREIGN KEY (`permission_id`)
		REFERENCES `survey_of_experts`.`permission` (`id`)
		ON DELETE NO ACTION
		ON UPDATE NO ACTION
)ENGINE = InnoDB;


-- -------------------------------------------------------

-- ---------------Структура таблиці quiz-------------------
DROP TABLE IF EXISTS `survey_of_experts`.`quiz` ;

CREATE TABLE IF NOT EXISTS `survey_of_experts`.`quiz` (
    `id` INT NOT NULL AUTO_INCREMENT,
    `type` TEXT(200),
    `topic` TEXT(200),
    `text` TEXT(200),
    `date` DATETIME,
    PRIMARY KEY (`id`)
);

-- -------------------------------------------------------

-- ---------------Структура таблиці question-------------------
DROP TABLE IF EXISTS `survey_of_experts`.`question` ;

CREATE TABLE IF NOT EXISTS `survey_of_experts`.`question` (
    `id` INT NOT NULL AUTO_INCREMENT,
    `type` TEXT(200),
    `text` TEXT(200),
    `topic` TEXT(200),
    `quiz_id` INT NOT NULL,
    `answer_id` INT NOT NULL,
    PRIMARY KEY (`id`),
    CONSTRAINT `fk_question_quiz`
		FOREIGN KEY (`quiz_id`)
		REFERENCES `survey_of_experts`.`quiz` (`id`)
		ON DELETE NO ACTION
		ON UPDATE NO ACTION,
	CONSTRAINT `fk_question_answer`
		FOREIGN KEY (`answer_id`)
		REFERENCES `survey_of_experts`.`answer` (`id`)
		ON DELETE NO ACTION
		ON UPDATE NO ACTION
)ENGINE = InnoDB;

-- -------------------------------------------------------

-- ---------------Структура таблиці option-------------------
DROP TABLE IF EXISTS `survey_of_experts`.`option_` ;

CREATE TABLE IF NOT EXISTS `survey_of_experts`.`option_`(
    `id` INT NOT NULL AUTO_INCREMENT,
    `type` TEXT(200),
    `text` TEXT(200),
    `question_id` INT NOT NULL,
    PRIMARY KEY (`id`),
    CONSTRAINT `fk_option_question`
		FOREIGN KEY (`question_id`)
		REFERENCES `survey_of_experts`.`question` (`id`)
		ON DELETE NO ACTION
		ON UPDATE NO ACTION
)ENGINE = InnoDB;

-- -------------------------------------------------------

-- ---------------Структура таблиці selectedOption-------------------
DROP TABLE IF EXISTS `survey_of_experts`.`selectedOption` ;

CREATE TABLE IF NOT EXISTS `survey_of_experts`.`selectedOption`(
    `id` INT NOT NULL AUTO_INCREMENT,
    `option_id` INT NOT NULL,
    PRIMARY KEY (`id`),
    CONSTRAINT `fk_selectedOption_option`
		FOREIGN KEY (`option_id`)
		REFERENCES `survey_of_experts`.`option_` (`id`)
		ON DELETE NO ACTION
		ON UPDATE NO ACTION
)ENGINE = InnoDB;

-- -------------------------------------------------------


-- ---------------Структура таблиці answer-------------------
DROP TABLE IF EXISTS `survey_of_experts`.`answer` ;

CREATE TABLE IF NOT EXISTS `survey_of_experts`.`answer` (
    `id` INT NOT NULL AUTO_INCREMENT,
    `text` TEXT(200) NOT NULL,
    `data` DATE NOT NULL,
    `user_id` INT NOT NULL,
    `selectedOption_id` INT NOT NULL,
    PRIMARY KEY (`id`),
	CONSTRAINT `fk_answer_user`
		FOREIGN KEY (`user_id`)
		REFERENCES `survey_of_experts`.`user` (`id`)
		ON DELETE NO ACTION
		ON UPDATE NO ACTION,
	CONSTRAINT `fk_answer_selectedOption`
		FOREIGN KEY (`selectedOption_id`)
		REFERENCES `survey_of_experts`.`selectedOption` (`id`)
		ON DELETE NO ACTION
		ON UPDATE NO ACTION
);

# Налаштування Express сервера

const express = require('express');
const cors = require('cors');
const router = require('./routes');
const AppError = require('./utils/appError');
const errorHandler = require('./utils/errorHandler');

const app = express();

app.use(express.json());
app.use(cors());
app.use('/api', router);

app.all('*', (req, res, next) => {
  next(new AppError(`The URL ${req.originalUrl} does not exists`, 404));
});
app.use(errorHandler);

const PORT = 3000;
app.listen(PORT, () => {
  console.log(`server running on port ${PORT}`);
});

module.exports = app;

# Створення контролерів додатка

const AppError = require('../utils/appError');
const conn = require('../service/db');

exports.getAllUsers = (req, res, next) => {
  conn.query('SELECT * FROM user', function (err, data, fields) {
    if (err) return next(new AppError(err));
    res.status(200).json({
      status: 'success',
      length: data?.length,
      data: data,
    });
  });
};

exports.getUserById = (req, res, next) => {
  if (!req.params.id) {
    return next(new AppError('No user id found', 404));
  }
  conn.query(
    'SELECT * FROM user WHERE id = ?',
    [req.params.id],
    function (err, data, fields) {
      if (err) return next(new AppError(err, 500));
      res.status(200).json({
        status: 'success',
        length: data?.length,
        data: data,
      });
    }
  );
};

exports.createUser = (req, res, next) => {
  if (!req.body) return next(new AppError('No form data found', 404));
  const values = [
    req.body.username,
    req.body.email,
    req.body.password,
    req.body.permission_id,
  ];
  conn.query(
    'INSERT INTO user (username, email, password, permission_id) VALUES(?)',
    [values],
    function (err, data, fields) {
      if (err) return next(new AppError(err, 500));
      res.status(201).json({
        status: 'success',
        message: 'user added!',
      });
    }
  );
};

exports.deleteUser = (req, res, next) => {
  if (!req.params.id) {
    return next(new AppError('No todo id found', 404));
  }
  conn.query(
    'DELETE FROM user WHERE id=?',

      if (err) return next(new AppError(err, 500));

};

# Створення глобальних обробників помилок

class AppError extends Error {
  constructor(msg, statusCode) {
    super(msg);

    this.statusCode = statusCode;
    this.error = `${statusCode}`.startsWith('4') ? 'fail' : 'error';
    this.isOperational = true;

    Error.captureStackTrace(this, this.constructor);
  }
}

module.exports = AppError;

module.exports = (err, req, res, next) => {
  err.statusCode = err.statusCode || 500;
  err.status = err.status || 'error';
  res.status(err.statusCode).json({
    status: err.status,
    message: err.message,
  });
};
